        // Export for GitHub
        function exportForGitHub() {
            const title = titleInput.value || "–°—Ç–∞—Ç—Ç—è";
            const description = descriptionInput.value || "–û–ø–∏—Å —Å—Ç–∞—Ç—Ç—ñ";
            const status = statusSelect.value;
            const slug = generateSlug(title);

            // Generate JSON
            const jsonData = {
                "title": title,
                "description": description,
                "image": "script.jpg",
                "link": `./scripts/${slug}.html`,
                "status": status
            };

            const jsonString = JSON.stringify(jsonData, null, 2);
            document.getElementById("jsonOutput").textContent = jsonString;

            // Generate HTML file
            const htmlContent = generateGitHubHTML();
            const blob = new Blob([htmlContent], { type: "text/html" });
            const url = URL.createObjectURL(blob);
            const downloadLink = document.getElementById("downloadHtmlLink");
            downloadLink.href = url;
            downloadLink.download = `${slug}.html`;

            // Show modal
            document.getElementById("githubModal").style.display = "block";
        }

        // Close modals
        function closeGitHubModal() {
            document.getElementById("githubModal").style.display = "none";

        // Close modal when clicking outside
        window.onclick = function(event) {
            const githubModal = document.getElementById("githubModal");
            const previewModal = document.getElementById("previewModal");
            if (event.target == githubModal) {
                githubModal.style.display = "none";
            }
            if (event.target == previewModal) {
                previewModal.style.display = "none";
